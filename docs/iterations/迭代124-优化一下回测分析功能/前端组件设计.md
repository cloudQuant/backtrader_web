## 1. 路由设计 [P0]

在 `src/frontend/src/router/index.ts` 新增路由:

```typescript
// 工作区路由
{
  path: '/workspace',
  name: 'WorkspaceList',
  component: () => import('@/views/workspace/WorkspaceListPage.vue'),
  meta: { title: '策略研究', requiresAuth: true }
},
{
  path: '/workspace/:id',
  name: 'WorkspaceDetail',
  component: () => import('@/views/workspace/WorkspaceDetailPage.vue'),
  meta: { title: '工作区', requiresAuth: true },
  children: [
    {
      path: '',
      name: 'WorkspaceUnits',
      component: () => import('@/views/workspace/WorkspaceUnitsTab.vue')
    },
    {
      path: 'optimization/:unitId',
      name: 'WorkspaceOptimization',
      component: () => import('@/views/workspace/WorkspaceOptimizationTab.vue')
    },
    {
      path: 'report/:reportId?',
      name: 'WorkspaceReport',
      component: () => import('@/views/workspace/WorkspaceReportTab.vue')
    }
  ]
}
```

侧边栏导航新增"策略研究"入口（在"回测分析"下方）。

### 7.1 P0 首版：轮询

```typescript
// 在 WorkspaceDetailPage.vue 中
let pollTimer: ReturnType<typeof setInterval> | null = null

function startPolling(workspaceId: string) {
  pollTimer = setInterval(async () => {
    const status = await workspaceApi.getRunStatus(workspaceId)
    // 更新 units 状态
    store.updateUnitsStatus(status.units)
  }, 3000) // 3秒轮询
}

function stopPolling() {
  if (pollTimer) clearInterval(pollTimer)
}
```

```python
def parse_optimize_setting(setting: str) -> dict:
    """解析优化设置字符串为 param_range"""
    if not setting:
        return None
    
    parts = setting.split(':')
    method = parts[0]  # 等差/等比/枚举
    values = parts[1].split(',')
    
    if method == '等差' or method == 'equal_diff':
        return {
            'start': float(values[0]),
            'end': float(values[1]),
            'step': float(values[2]),
            'type': 'float'
        }
    elif method == '等比' or method == 'equal_ratio':
        return {
            'start': float(values[0]),
            'end': float(values[1]),
            'ratio': float(values[2]),
            'type': 'geometric'
        }
    elif method == '枚举' or method == 'enum':
        return {
            'values': [float(v) for v in values],
            'type': 'enum'
        }
```

```python
class ExtendedMetrics:
    """TBQuant 完整指标集"""
    
    # 基础
    initial_cash: float         # 初始资金
    net_value: float            # 净值 = final_value / initial_cash
    net_profit: float           # 净利润 = final_value - initial_cash

    # 风险调整指标
    sharpe_ratio: float         # 夏普比率
```

```python
class ExtendedMetricsService:
    """扩展指标计算服务"""
    
    def calculate(self, equity_curve, trades, daily_returns=None) -> dict:
        """计算完整的TBQuant指标集"""
        metrics = {}
        
        # 基础指标
        metrics['initial_cash'] = equity_curve[0] if equity_curve else 0
        metrics['net_value'] = equity_curve[-1] / equity_curve[0] if equity_curve else 1
        metrics['net_profit'] = equity_curve[-1] - equity_curve[0] if equity_curve else 0
        
        # 年化收益率（使用 fincore 库）
        metrics['annual_return'] = self._calc_annual_return(equity_curve)
        
        # 回撤指标
        dd_value, dd_rate = self._calc_max_drawdown(equity_curve)
        metrics['max_drawdown_value'] = dd_value
        metrics['max_drawdown_rate'] = dd_rate
        
        # 夏普比率
        metrics['sharpe_ratio'] = self._calc_sharpe(daily_returns)
        
        # 调整收益风险比
        metrics['adjusted_return_risk'] = (
            metrics['annual_return'] / dd_rate if dd_rate > 0 else 0
        )
        
        # 交易统计
        metrics.update(self._calc_trade_stats(trades))
        
        # 日/周/月统计
        metrics.update(self._calc_period_stats(daily_returns, 'daily'))
        metrics.update(self._calc_period_stats(
            self._aggregate_returns(daily_returns, 'weekly'), 'weekly'))
        metrics.update(self._calc_period_stats(
            self._aggregate_returns(daily_returns, 'monthly'), 'monthly'))
        
        return metrics
    
    def _calc_trade_stats(self, trades) -> dict:
        """计算交易统计指标"""
        if not trades:
            return {...}
        
        wins = [t for t in trades if t.get('pnl', 0) > 0]
        losses = [t for t in trades if t.get('pnl', 0) < 0]
        
        total_profit = sum(t['pnl'] for t in wins)
        total_loss = sum(t['pnl'] for t in losses)
        
        return {
            'trade_count': len(trades),
            'win_rate': len(wins) / len(trades) * 100 if trades else 0,
            'avg_profit': (total_profit + total_loss) / len(trades) if trades else 0,
            'total_profit': total_profit,
            'total_loss': total_loss,
            'profit_loss_ratio': abs(total_profit / total_loss) if total_loss else 0,
            'profit_factor': total_profit / abs(total_loss) if total_loss else 0,
            'win_odds': (len(wins)/len(trades)) * (abs(total_profit/total_loss)) if total_loss and trades else 0,
            'trade_cost': sum(t.get('commission', 0) for t in trades),
        }
    
    def _calc_period_stats(self, returns, prefix) -> dict:
        """计算某个周期级别的统计"""
        if not returns:
            return {f'{prefix}_avg_return': 0, f'{prefix}_max_loss': 0, f'{prefix}_max_profit': 0}
        
        return {
            f'{prefix}_avg_return': np.mean(returns) * 100,
            f'{prefix}_max_loss': min(returns) * 100,
            f'{prefix}_max_profit': max(returns) * 100,
        }
```

### 2.3 组合指标计算逻辑

```python
class ReportCalculationService:
    """组合报告计算服务"""
    
    def calculate_report(self, unit_results: list[dict], config: dict) -> dict:
        """计算多任务组合报告"""
        
        # 1. 裁剪每个单元的equity_curve到统计时间范围
        start = config['stat_start_date']
        end = config['stat_end_date']
        
        trimmed_curves = []
        for result in unit_results:
            curve = self._trim_curve(result['equity_curve'], start, end)

        max_cash = config['max_invest_cash'] * 10000  # 万→元
        
        combined_curve = self._combine_equity_curves(
            trimmed_curves, weights, max_cash
        )
        
        # 4. 计算组合指标
        combined_trades = []
        for result in unit_results:
            combined_trades.extend(result['trades'])
        combined_trades.sort(key=lambda t: t.get('datetime', ''))
```

### 3.2 列配置持久化

- 优化结果: `unit_optimization_results` 或 `localStorage`
- 组合报告: `workspace_reports.custom_fields` 或 `localStorage`

推荐: 使用 `localStorage` 存储前端展示偏好，不增加后端复杂度。

```typescript
// 持久化列配置
const STORAGE_KEY = 'workspace_optimization_columns'

function saveColumnConfig(columns: string[]) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(columns))
}

function loadColumnConfig(): string[] | null {
  const saved = localStorage.getItem(STORAGE_KEY)
  return saved ? JSON.parse(saved) : null
}
```

> **文档状态**: 初版  
> **最后更新**: 2026-04-13  
> **核心策略**: 最大化复用策略研究组件，通过 props/slot 差异化交易行为

## 1. 设计原则

1. **复用优先** — 策略交易页面复用策略研究的 WorkspaceListPage、WorkspaceDetailPage、WorkspaceUnitsTab 等组件
2. **差异化通过 Props** — 通过 `workspaceType: 'trading'` prop 控制组件行为差异
3. **新增组件最小化** — 只在交易专属功能（自动交易调度、头寸管理器、锁定按钮）时新建组件
4. **样式一致** — 与策略研究保持完全一致的 Element Plus + TailwindCSS 样式

#### 4.3.4 辅助函数

```typescript
function getTradingField(row: StrategyUnit, field: string): any {
  // 优先从运行时状态获取，降级到快照
  return row.trading_status?.[field]
    ?? row.trading_snapshot?.[field]
    ?? null
}

function pnlColorClass(val: number | null): string {
  if (val == null || val === 0) return 'text-gray-500'
  return val > 0 ? 'text-red-500' : 'text-green-500'
}

function formatPnl(val: number | null): string {
  if (val == null) return '-'
  return val >= 0 ? `+${val.toFixed(2)}` : val.toFixed(2)
}

function formatPrice(val: number | null): string {
  if (val == null || val === 0) return '-'
  return val.toFixed(2)
}

function formatPercent(val: number | null): string {
  if (val == null) return '-'
  return `${val >= 0 ? '+' : ''}${val.toFixed(2)}%`
}

function formatMoney(val: number | null): string {
  if (val == null || val === 0) return '-'
  return val.toLocaleString('zh-CN', { maximumFractionDigits: 0 })
}

function tradingStatusLabel(row: StrategyUnit): string {
  if (row.lock_trading && row.lock_running) return '全锁定'
  if (row.lock_trading) return '交易锁'
  if (row.lock_running) return '运行锁'
  const status = row.trading_status?.is_running ?? row.run_status === 'running'
  return status ? '运行中' : '已停止'
}

function tradingStatusType(row: StrategyUnit): string {
  if (row.lock_trading || row.lock_running) return 'warning'
  const running = row.trading_status?.is_running ?? row.run_status === 'running'
  return running ? 'success' : 'info'
}
```

async function startTrading(workspaceId: string, unitIds: string[], parallel = false) {

}

}
